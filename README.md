# FleakOS
FleakOS Kernel Source Tree

# Dependencias

> sudo apt-get install -y xorriso

> sudo apt-get install -y gcc-multilib

> sudo apt-get install -y nasm

> sudo apt install -y qemu-system-x86

> apt-get install grub-common

# Executando 

> chmod +x build.sh

> ./build.sh

# Removendo lixo

> chmod +x remove.sh

> ./remove.sh
